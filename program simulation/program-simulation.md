# `for` loop

```cpp
int main() {
    for(int start = 0; start < 10; start++) {
        cout << "My name is Alexandra Lanorias." << endl;
    }

    return 0;
}
```

# `while` loop

```cpp
int main() {
    int start = 0;

    while(start < 10) {
        cout << "We are now using a while loop." << endl;
        start++;
    }

    return 0;
}
```

# `do while` loop

```cpp
int main() {
    int start = 0;

    do {
        cout << "We are now using a do while loop." << endl;
        start++;
    } while(start < 10);

    return 0;
}
```
